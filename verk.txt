// 7. 
// Convert a whole number to a string in any base (decimal, binary, and so on) 
// a) Reyndu að greina kóðann t.d. með að setja console.log() í kóðann til að fá frekari upplýsingar
// b) Notaðu debugger í chromeDeveloper eða firebug. (breakpoint á ákveðnum línum til að geta skoðað gildi)
function numberToString(n, base) {
  debugger;
  var result = "", sign = "";
  if (n < 0) {
    sign = "-";
    n = -n;
  }
  do {
    result = String(n % base) + result;
    n /= base;                                 
  } while (n > 0);
  return sign + result;
}
console.log(numberToString(13, 10)); // ? 1.5e-3231.3e-3221.3e-3211.3e-3201.3e-3191.3e-3181.3…


// 8. 
// Útskýrðu notkun á isNaN í kóðanum, afhverju að gera þetta?
function promptNumber(question) {
  var result = Number(prompt(question, ""));
  if (isNaN(result)) return null;//isNaN þýðir "is Not a Number" svo hér er verið að kanna hvort að þetta sé ekki tala og ef þetta er ekki taka skilar hann null
  else return result;
}

console.log(promptNumber("How many trees do you see?")); //hér prentast út tala sme þú slærð inn eða null ef þú slærð ekki inn tölu


// 9. 
//  Útskýrðu hvernig try og catch virkar hér í kóðanum, 
//  hvað gerir throw keyword og hvað gerist þegar það verður error (útskýra kóðaflæði)

function promptDirection(question) {
  var result = prompt(question, "");
  if (result.toLowerCase() == "left") return "L";
  if (result.toLowerCase() == "right") return "R";
  throw new Error("Invalid direction: " + result);  //hér er notað throw til þess að kasta villu
}

function look() {
  if (promptDirection("Which way?") == "L")
    return "a house";
  else
    return "two angry bears";
}

try {
  console.log("You see", look()); //hér er kalla ð í look og hún getur kastað villu
} catch (error) {
  console.log("Something went wrong: " + error);//hér eru við búin að grípa villu ef henni er kastað og textinn prentaður út 
}
